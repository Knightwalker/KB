<?php

$n1 = readline();
$n2 = readline();
$n3 = readline();

for($a = 1; $a <= $n1; $a++) {
    for($b = 2; $b <= $n2; $b++) {
        for($c = 1; $c <= $n3; $c++) { 
       
            if ($a % 2 == 0 && $c % 2 == 0) {
                
                if ($b == 2 || $b == 3 || $b == 5 || $b == 7) {
                    echo $a . " " . $b . " " . $c . "\n";
                }
            }     
        }     
        
    }   
}
